-- Funﾃｧﾃ｣o para Fling School Bus
local function FlingSchoolBus(targetPlayer)
    local LocalPlayer = game.Players.LocalPlayer
    local RS = game:GetService("ReplicatedStorage")
 
    if targetPlayer.Character and LocalPlayer.Character then
        local playerChar = LocalPlayer.Character
        local playerHumanoid = playerChar:FindFirstChildOfClass("Humanoid")
        local playerHRP = playerChar:FindFirstChild("HumanoidRootPart")
 
        if playerHumanoid and playerHRP then
            local originalPosition = playerHRP.CFrame 
 
            pcall(function()
                playerHRP.Anchored = true
                playerHRP.CFrame = CFrame.new(1181.83, 76.08, -1158.83)
                task.wait(0.2)
                playerHRP.Velocity, playerHRP.RotVelocity = Vector3.new(0,0,0), Vector3.new(0,0,0)
                playerHRP.Anchored = false
                playerHumanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
            end)
 
            task.wait(0.5)
 
            pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("DeleteAllVehicles") end)
            pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("PickingCar", "SchoolBus") end)
 
            task.wait(1)
 
            local vFolder = game.Workspace:FindFirstChild("Vehicles")
            if vFolder then
                local bus = vFolder:FindFirstChild(LocalPlayer.Name .. "Car")
                if bus then
                    pcall(function()
                        playerHRP.Anchored = true
                        playerHRP.CFrame = CFrame.new(1171.15, 79.45, -1166.2)
                        task.wait(0.2)
                        playerHRP.Velocity, playerHRP.RotVelocity = Vector3.new(0,0,0), Vector3.new(0,0,0)
                        playerHRP.Anchored = false
                        playerHumanoid:ChangeState(Enum.HumanoidStateType.Seated)
                    end)
 
                    local startTime = tick()
                    while true do
                        task.wait()
                        if tick() - startTime > 10 then break end
 
                        if playerHumanoid.Sit then
                            for _, part in ipairs(bus:GetDescendants()) do
                                if part:IsA("BasePart") then
                                    part.CanCollide = false
                                    pcall(function() part:SetNetworkOwner(nil) end)
                                end
                            end
 
                            local running = true
                            local connection = game:GetService("RunService").Stepped:Connect(function()
                                if running then
                                    for _, part in ipairs(playerChar:GetDescendants()) do 
                                        if part:IsA("BasePart") then part.CanCollide = false end 
                                    end
                                end
                            end)
 
                            local flingStart = tick()
                            local flingConnection
                            flingConnection = game:GetService("RunService").Heartbeat:Connect(function()
                                if not running or not (targetPlayer and targetPlayer.Character) then 
                                    running = false 
                                    return 
                                end
 
                                local tHRP = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                                local tHum = targetPlayer.Character:FindFirstChild("Humanoid")
 
                                if not (tHRP and tHum and playerHRP and playerHumanoid) then 
                                    running = false 
                                    return 
                                end
 
                                pcall(function()
                                    bus:PivotTo(CFrame.new(tHRP.Position + Vector3.new(math.random(-10,10),0,math.random(-10,10))) * CFrame.Angles(math.rad(game.Workspace.DistributedGameTime*12000),math.rad(game.Workspace.DistributedGameTime*15000),math.rad(game.Workspace.DistributedGameTime*18000)))
                                end)
 
                                local targetSitting = false
                                for _, p in ipairs(bus:GetDescendants()) do
                                    if (p:IsA("Seat") or p:IsA("VehicleSeat")) and p.Name ~= "VehicleSeat" and p.Occupant == tHum then 
                                        targetSitting = true 
                                        break 
                                    end
                                end
 
                                if targetSitting or tick() - flingStart > 10 then
                                    running = false
                                    flingConnection:Disconnect()
 
                                    pcall(function() bus:PivotTo(CFrame.new(-59599.73, 2040070.5, -293391.16)) end)
 
                                    local bv = Instance.new("BodyVelocity")
                                    bv.MaxForce = Vector3.new(math.huge,math.huge,math.huge)
                                    bv.Velocity = Vector3.new(0,1000000,0)
                                    bv.Parent = playerHRP
 
                                    task.wait(0.5)
                                    pcall(function() RS:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("DeleteAllVehicles") end)
                                    if bv then bv:Destroy() end
 
                                    for _, p in ipairs(playerChar:GetDescendants()) do 
                                        if p:IsA("BasePart") then p.CanCollide = true end 
                                    end
 
                                    pcall(function()
                                        playerHRP.Anchored = true
                                        playerHRP.CFrame = originalPosition
                                        task.wait(0.2)
                                        playerHRP.Velocity, playerHRP.RotVelocity = Vector3.new(0,0,0), Vector3.new(0,0,0)
                                        playerHRP.Anchored = false
                                        playerHumanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
                                    end)
                                    connection:Disconnect()
                                end
                            end)
                            return
                        end
                    end
                end
            end
        end
    end
end

-- Funﾃｧﾃ｣o para Fling - Boat (extraﾃｭda do botﾃ｣o "Fling - Boat")
local function FlingBoat(targetPlayer)
    if not targetPlayer or not targetPlayer.Character then
        warn("Jogador alvo invﾃ｡lido")
        return
    end

    local Player = game.Players.LocalPlayer
    local Character = Player.Character
    local Humanoid = Character and Character:FindFirstChildOfClass("Humanoid")
    local RootPart = Character and Character:FindFirstChild("HumanoidRootPart")
    local Vehicles = game.Workspace:FindFirstChild("Vehicles")

    if not Humanoid or not RootPart then
        warn("Humanoid ou RootPart invﾃ｡lido")
        return
    end

    local function spawnBoat()
        RootPart.CFrame = CFrame.new(1754, -2, 58)
        task.wait(0.5)
        game:GetService("ReplicatedStorage").RE:FindFirstChild("1Ca1r"):FireServer("PickingBoat", "MilitaryBoatFree")
        task.wait(1)
        return Vehicles:FindFirstChild(Player.Name.."Car")
    end

    local PCar = Vehicles:FindFirstChild(Player.Name.."Car") or spawnBoat()
    if not PCar then
        warn("Falha ao spawnar o barco")
        return
    end

    local Seat = PCar:FindFirstChild("Body") and PCar.Body:FindFirstChild("VehicleSeat")
    if not Seat then
        warn("Assento nﾃ｣o encontrado")
        return
    end

    local function GetBall()
        if not backpack:FindFirstChild("SoccerBall") and not character:FindFirstChild("SoccerBall") then

        end

        local ballName = "Soccer" .. player.Name
        local foundBall = nil
        local timeout = tick()
 
        repeat 
            task.wait() 
            foundBall = ServerBalls:FindFirstChild(ballName)
        until foundBall or (tick() - timeout > 5)
 
        if tool.Parent == character then tool.Parent = backpack end
        return foundBall
    end

    if not Ball then 
        warn("Nﾃ｣o foi possﾃｭvel obter a bola.")

    Ball.CanCollide = false
    Ball.Massless = true
    Ball.CustomPhysicalProperties = PhysicalProperties.new(100, 0, 0, 100, 100)
 
    if target and target ~= player then
        local tchar = target.Character
        if tchar and tchar:FindFirstChild("HumanoidRootPart") and tchar:FindFirstChild("Humanoid") then
            local troot = tchar.HumanoidRootPart
            local thum = tchar.Humanoid
 
            for _, v in pairs(Ball:GetChildren()) do
                if v:IsA("BodyVelocity") or v:IsA("BodyForce") or v:IsA("BodyAngularVelocity") then v:Destroy() end
            end
 
            local bv = Instance.new("BodyVelocity")
            bv.Name = "FlingPower"
            bv.Velocity = Vector3.new(9e8, 9e8, 9e8)
            bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
            bv.P = 9e9

            workspace.CurrentCamera.CameraSubject = thum
 
            local connection
            connection = game:GetService("RunService").Heartbeat:Connect(function()
                if not Ball or not Ball.Parent or not troot or not troot.Parent or thum.Health <= 0 then
                    connection:Disconnect()
                    workspace.CurrentCamera.CameraSubject = humanoid
                    return
                end
 
                -- PREDIﾃﾃグ DE MOVIMENTO (MELHORADO PARA CARROS)
                if troot.Velocity.Magnitude > 0.1 then
                    local Prediction = troot.Velocity * 0.12
                    Ball.CFrame = CFrame.new(troot.Position + Prediction)
                else
                    Ball.CFrame = troot.CFrame
                end
                
                -- MANTﾃ窺 O GIRO CONSTANTE
                Ball.RotVelocity = Vector3.new(10000, 10000, 10000)
            end)
 
            task.delay(30, function()
                if connection then connection:Disconnect() end
            end)
        end
    end
end

local function spawnPirateShip()
    local character = LocalPlayer.Character
    if not character then return nil end
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    
    hrp.CFrame = CFrame.new(1754, -2, 58)
    task.wait(0.5)
    ReplicatedStorage:WaitForChild("RE"):FindFirstChild("1Ca1r"):FireServer("PickingBoat", "PirateFree")
    task.wait(1)
    return workspace.Vehicles:FindFirstChild(LocalPlayer.Name .. "Car")
end

local function startTornado()
    ShipSystem.currentShip = spawnPirateShip()
    if not ShipSystem.currentShip then return end
    
    local seat = ShipSystem.currentShip:FindFirstChild("Body")
    seat = seat and seat:FindFirstChild("VehicleSeat")
    local character = LocalPlayer.Character
    local hum = character and character:FindFirstChild("Humanoid")
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    
    if not seat or not hum then return end
    
    repeat
        task.wait(0.1)
        hrp.CFrame = seat.CFrame * CFrame.new(0, 1, 0)
    until hum.SeatPart == seat
    
    -- REMOVIDO playShipAudio
    
    task.delay(4, function()
        if hum.SeatPart then hum.Sit = false end
        hrp.CFrame = CFrame.new(0, 0, 0)
    end)
    
    ShipSystem.tornadoActive = true
    
    task.spawn(function()
        local re = ReplicatedStorage:WaitForChild("RE"):WaitForChild("1Player1sCa1r")
        while ShipSystem.tornadoActive and ShipSystem.currentShip and ShipSystem.currentShip.Parent do
            re:FireServer("Flip")
            task.wait(0.5)
        end
    end)
    
    local rotationSpeed = math.rad(720)
    local currentRotation = 0
    
    local connection
    connection = RunService.Heartbeat:Connect(function(dt)
        if not ShipSystem.tornadoActive or not ShipSystem.currentShip or not ShipSystem.currentShip.PrimaryPart then
            connection:Disconnect()
            return
        end
        
        currentRotation = currentRotation + rotationSpeed * dt
        
        -- USANDO A Lﾃ敵ICA DE ALTURA DO SEU SCRIPT ORIGINAL (0 NO VETOR)
        local cframe = CFrame.new(localSelecionado) * CFrame.Angles(0, currentRotation, 0)
        ShipSystem.currentShip:SetPrimaryPartCFrame(cframe)
    end)
end

local function stopTornado()
    ShipSystem.tornadoActive = false
    task.spawn(function()
        ReplicatedStorage:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer("DeleteAllVehicles")
    end)
end

local function ApplyVehicleCollision(vehicle)
    local vehicleTypes = {
        "Shark", "WaveRunner", "SailBoatFree", "SpeedBoatFree",
        "PoliceBoatChangeable", "MilitaryBoatFree", "PirateFree",
        "Jet", "Helicopter"
    }
    
    for _, part in pairs(vehicle:GetDescendants()) do
        if part:IsA("BasePart") then
            local partName = part.Name:lower()
            local isWaterVehicle = false
            
            for _, vType in ipairs(vehicleTypes) do
                if vehicle.Name == vType then
                    isWaterVehicle = true
                    break
                end
            end
            
            part.CanCollide = isWaterVehicle and (
                part:IsA("Seat") or part:IsA("VehicleSeat") or
                partName:find("wheel") or partName:find("tire") or
                partName:find("chassis") or partName:find("base") or
                partName:find("frame") or partName:find("hull") or
                partName:find("propeller") or partName:find("rotor") or
                partName:find("wing")
            ) or (
                part:IsA("Seat") or part:IsA("VehicleSeat") or
                partName:find("wheel") or partName:find("tire") or
                partName:find("chassis") or partName:find("base") or
                partName:find("frame")
            )
        end
    end
end

local antiLagAtivo = false

-- Itens que serﾃ｣o monitorados
local itensParaLimitar = {"Laptop", "FireX", "Bomb", "Stretcher"}

local function MonitorContainer(container)
    container.ChildAdded:Connect(function(child)
        -- Se o Anti-Lag estiver ON e o item estiver na lista negra
        if antiLagAtivo and table.find(itensParaLimitar, child.Name) then
            local count = 0
            for _, item in ipairs(container:GetChildren()) do
                if item.Name == child.Name then
                    count = count + 1
                    -- Se passar de 10, remove o item
                    if count > 10 then
                        item.Parent = nil
                    end
                end
            end
        end
    end)
end

local function SetupAntiLagForPlayer(player)
    -- REGRA: Se o jogador for VOCﾃ, o script nﾃ｣o monitora nada (vocﾃｪ fica sem limite)
    if player == LocalPlayer then return end

    local function processPlayer(p)
        local backpack = p:FindFirstChild("Backpack")
        if backpack then MonitorContainer(backpack) end
        
        if p.Character then MonitorContainer(p.Character) end
        
        p.CharacterAdded:Connect(function(char)
            MonitorContainer(char)
        end)
    end
    processPlayer(player)
end

-- // ｧ Lﾃ敵ICA CENTRAL DE MOVIMENTO (Mantenha este bloco no final do script)
local angle = 0
RunService.Heartbeat:Connect(function()
    local char = LPlayer.Character
    local hum = char and char:FindFirstChild("Humanoid")
    local seat = hum and hum.SeatPart

    if seat and seat:IsA("VehicleSeat") then
        local bv = seat:FindFirstChild("OmniBV") or Instance.new("BodyVelocity", seat)
        bv.Name = "OmniBV"
        bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
        
        local bg = seat:FindFirstChild("OmniBG") or Instance.new("BodyGyro", seat)
        bg.Name = "OmniBG"
        bg.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)

        local avoidForce = Vector3.new(0, 0, 0)
        if _G.AvoidObstacles then
            local ray = workspace:Raycast(seat.Position, seat.CFrame.LookVector * 20, RaycastParams.new())
            if ray and not ray.Instance:IsDescendantOf(char) then
                avoidForce = Vector3.new(0, 45, 0)
            end
        end

        if _G.CarFlyEnabled then
            bg.CFrame = workspace.CurrentCamera.CFrame
            local dir = Vector3.new(0,0,0)
            if seat.Throttle == 1 then dir = dir + workspace.CurrentCamera.CFrame.LookVector end
            if seat.Throttle == -1 then dir = dir - workspace.CurrentCamera.CFrame.LookVector end
            if seat.Steer == 1 then dir = dir + workspace.CurrentCamera.CFrame.RightVector end
            if seat.Steer == -1 then dir = dir - workspace.CurrentCamera.CFrame.RightVector end
            bv.Velocity = (dir * _G.FlySpeed) + avoidForce
            if dir.Magnitude == 0 then bv.Velocity = Vector3.new(0, 0.1, 0) end

        elseif _G.OrbitEnabled and _G.TargetName ~= "" and _G.TargetName ~= "Ninguﾃｩm" then
            local target = Players:FindFirstChild(_G.TargetName)
            if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                local tHRP = target.Character.HumanoidRootPart
                local dist = (seat.Position - tHRP.Position).Magnitude
                if dist > 15 then
                    bv.Velocity = ((tHRP.Position - seat.Position).Unit * _G.FlySpeed) + avoidForce
                    bg.CFrame = CFrame.new(seat.Position, tHRP.Position)
                else
                    angle = angle + 0.05
                    local orbitPos = tHRP.Position + Vector3.new(math.cos(angle)*12, 8, math.sin(angle)*12)
                    bv.Velocity = (orbitPos - seat.Position) * 10
                    bg.CFrame = CFrame.new(seat.Position, tHRP.Position)
                end
            end
        else
            bv.MaxForce = Vector3.new(0,0,0)
            bg.MaxTorque = Vector3.new(0,0,0)
        end
    end
end)

-- Funﾃｧﾃ｣o para tocar o ﾃ｡udio repetidamente para TODOS os players
local function playLoopedAudio()
    while audioLoop do
        if selectedAudioID then
            local idStr = tostring(selectedAudioID)
            
            -- Loop para aplicar o remote em cada player do servidor
            for _, p in pairs(game:GetService("Players"):GetPlayers()) do
                if p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                    local hrp = p.Character.HumanoidRootPart
                    
                    local args = {
                        [1] = hrp,
                        [2] = hrp,
                        [3] = Vector3.new(0, -1, 0),
                        [4] = Vector3.new(0, 0, 0),
                        [5] = hrp,
                        [6] = hrp,
                        [7] = idStr,
                        [8] = idStr,
                        [9] = { [1] = true },
                        [10] = {
                            [1] = 25,
                            [2] = Vector3.new(0.25, 0.25, 100),
                            [3] = BrickColor.new("Bright yellow"),
                            [4] = 0.25,
                            [5] = Enum.Material.SmoothPlastic,
                            [6] = 0.25
                        },
                        [11] = true,
                        [12] = false
                    }
                    
                    -- Dispara o remote para o player atual do loop
                    pcall(function()
                        game:GetService("ReplicatedStorage"):WaitForChild("RE", 9e9):WaitForChild("1Gu1n", 9e9):FireServer(unpack(args))
                    end)
                end
            end
        else
            warn("Nenhum ﾃ｡udio selecionado!")
        end
 
        task.wait(0.5) -- Delay para evitar lag/kick por excesso de remote
    end
end

        if workspace:FindFirstChild(pl.Name) then
            for _, v in pairs(workspace[pl.Name]:GetChildren()) do

                if babyTorso and babyTorso:FindFirstChild("BodyPosition") and victimHRP then
                    -- Lﾃ敵ICA DA SARRADA:
                    -- O 'math.sin' faz o bebﾃｪ ir para frente e para trﾃ｡s rapidamente
                    local speed = 15 -- Velocidade do movimento
                    local distance = 0.8 -- Intensidade/Profundidade
                    local sarradaAnim = math.sin(tick() * speed) * distance
                    
                    -- Posiciona o bebﾃｪ nas costas/atrﾃ｡s da vﾃｭtima
                    babyTorso.BodyPosition.Position = victimHRP.Position + (victimHRP.CFrame.LookVector * (sarradaAnim - 1))
                    babyTorso.BodyGyro.CFrame = victimHRP.CFrame
                end
            end)
        end
    end
})

                local victimHRP = targetChar:FindFirstChild("HumanoidRootPart")
                local babyTorso = followChar:FindFirstChild("Torso")
                
                if babyTorso and victimHRP then
                    -- Lﾃ敵ICA DO GO GO GO:
                    -- Movimento rﾃ｡pido de vai e vem (cima/baixo e frente/trﾃ｡s)
                    local speed = 20 
                    local anim = math.sin(tick() * speed) * 0.5
                    
                    -- Posicionamento: 
                    -- Na frente do rosto (LookVector * 1.2)
                    -- Na altura da cabeﾃｧa (Y + 1.5)
                    local targetPos = victimHRP.CFrame * CFrame.new(0, 1.5 + (anim * 0.2), -1.2 + anim)
                    
                    babyTorso.BodyPosition.Position = targetPos.p
                    -- Faz o bebﾃｪ olhar para o player (invertido 180 graus)
                    babyTorso.BodyGyro.CFrame = victimHRP.CFrame * CFrame.Angles(0, math.rad(180), 0)
                end
            end)
        end
    end

local function SendChat(text)
    local chatService = game:GetService("TextChatService")
    if chatService.ChatVersion == Enum.ChatVersion.TextChatService then
        chatService.TextChannels.RBXGeneral:SendAsync(text)
    else
        game:GetService("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer(text, "All")
    end
end

-- Funﾃｧﾃ｣o para carregar animaﾃｧﾃｵes (GetObjects)
local function PlayCustomAnim(humanoid, id, priority, looped)
    local anim = Instance.new("Animation")
    pcall(function()
        local objs = game:GetObjects(id)
        if objs and #objs > 0 then id = objs[1].AnimationId end
    end)
    anim.AnimationId = id
    local track = humanoid:LoadAnimation(anim)
    track.Priority = priority or Enum.AnimationPriority.Action4
    track.Looped = looped or false
    track:Play()
    return track
end
